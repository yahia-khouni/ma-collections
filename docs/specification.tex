\section{Introduction}
% ============================================

\subsection{Contexte du Projet}
M\&A Collections est une plateforme e-commerce spécialisée dans la vente de vêtements et accessoires de mode premium en Tunisie. Le projet vise à offrir une expérience d'achat en ligne luxueuse et moderne, reflétant l'identité de marque haut de gamme.

\subsection{Objectifs}
\begin{itemize}
    \item Créer une boutique en ligne moderne avec une interface utilisateur élégante
    \item Offrir une expérience d'achat fluide et intuitive
    \item Permettre la gestion efficace des produits, commandes et clients
    \item Assurer la sécurité des transactions et des données personnelles
    \item Supporter le marché tunisien (devise TND, région Tunisie)
\end{itemize}

\subsection{Périmètre}
Le projet couvre:
\begin{itemize}
    \item Frontend (Storefront): Interface client avec Next.js 15
    \item Backend: API et administration avec Medusa.js v2
    \item Base de données: PostgreSQL
    \item Gestion des paiements et livraisons
\end{itemize}

\subsection{Fonctionnalités Principales}

\subsubsection{Module Catalogue}
\begin{itemize}
    \item \textbf{F1.1} - Affichage des produits avec pagination
    \item \textbf{F1.2} - Filtrage par catégories (Hommes, Femmes, T-Shirts, Chemises)
    \item \textbf{F1.3} - Tri des produits (prix, date, popularité)
    \item \textbf{F1.4} - Recherche de produits
    \item \textbf{F1.5} - Affichage détaillé d'un produit (images, variantes, prix)
    \item \textbf{F1.6} - Gestion des variantes (taille, couleur)
\end{itemize}

\subsubsection{Module Panier}
\begin{itemize}
    \item \textbf{F2.1} - Ajout de produits au panier
    \item \textbf{F2.2} - Modification des quantités
    \item \textbf{F2.3} - Suppression d'articles
    \item \textbf{F2.4} - Calcul automatique du total
    \item \textbf{F2.5} - Application de codes promotionnels
    \item \textbf{F2.6} - Persistance du panier (cookies/session)
\end{itemize}

\subsubsection{Module Authentification}
\begin{itemize}
    \item \textbf{F3.1} - Inscription avec email et mot de passe
    \item \textbf{F3.2} - Connexion sécurisée
    \item \textbf{F3.3} - Déconnexion
    \item \textbf{F3.4} - Réinitialisation du mot de passe
    \item \textbf{F3.5} - Gestion du profil utilisateur
\end{itemize}

\subsubsection{Module Compte Client}
\begin{itemize}
    \item \textbf{F4.1} - Tableau de bord personnel
    \item \textbf{F4.2} - Historique des commandes
    \item \textbf{F4.3} - Gestion des adresses de livraison
    \item \textbf{F4.4} - Modification des informations personnelles
    \item \textbf{F4.5} - Suivi des commandes en cours
\end{itemize}

\subsubsection{Module Commande (Checkout)}
\begin{itemize}
    \item \textbf{F5.1} - Saisie des informations de livraison
    \item \textbf{F5.2} - Sélection du mode de livraison
    \item \textbf{F5.3} - Choix du mode de paiement
    \item \textbf{F5.4} - Récapitulatif de commande
    \item \textbf{F5.5} - Confirmation et validation
    \item \textbf{F5.6} - Email de confirmation

\subsubsection{Module Navigation}

\section{Exigences Non Fonctionnelles}
% ============================================

\subsection{Performance}
\begin{itemize}
    \item Temps de chargement initial < 3 secondes
    \item Temps de réponse API < 500ms
    \item Support de 100+ utilisateurs simultanés
\end{itemize}

\subsection{Sécurité}
\begin{itemize}
    \item Authentification sécurisée (JWT)
    \item Protection CSRF et XSS
    \item Chiffrement des données sensibles
    \item HTTPS obligatoire
\end{itemize}

\subsection{Compatibilité}
\begin{itemize}
    \item Navigateurs: Chrome, Firefox, Safari, Edge (versions récentes)
    \item Responsive: Mobile, Tablette, Desktop
    \item Résolutions: 320px à 1920px+
\end{itemize}

\subsection{Design UI/UX}
\begin{itemize}
    \item Identité visuelle luxueuse (noir, or, blanc)
    \item Typographie élégante (Playfair Display, Inter)
    \item Animations fluides et transitions
    \item Accessibilité WCAG 2.1 niveau AA
\end{itemize}
